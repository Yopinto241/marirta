      const sections = [];
      // top-level sections: allow closing brace with optional comma
      const regex = /^\s*'([^']+)'\s*:\s*\{([\s\S]*?)^\s*\},?\s*$/gm;
      let m;
      while ((m = regex.exec(text)) !== null) {
        const name = m[1];
        const body = `'${name}': {` + m[2] + `\n},`;
        sections.push({ name, body });
